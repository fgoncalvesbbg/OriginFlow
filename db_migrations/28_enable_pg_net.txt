
-- 1. Enable the pg_net extension
-- This extension provides the net.http_post function used for webhooks and edge function calls from SQL triggers.
CREATE EXTENSION IF NOT EXISTS pg_net;

-- 2. Ensure the extension is properly initialized in the expected schema
-- Supabase usually handles this, but forcing it ensures the 'net' schema exists.
DO $$ 
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_namespace WHERE nspname = 'net') THEN
        CREATE SCHEMA IF NOT EXISTS net;
    END IF;
END $$;

-- 3. Grant permissions to use the extension functions
GRANT USAGE ON SCHEMA net TO postgres, authenticated, anon, service_role;
GRANT ALL ON ALL FUNCTIONS IN SCHEMA net TO postgres, authenticated, anon, service_role;

-- 4. Reload schema cache to make sure PostgREST sees the extension functions
NOTIFY pgrst, 'reload schema';
