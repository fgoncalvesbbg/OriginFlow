
-- 1. Create the Production Updates table (if it doesn't exist)
CREATE TABLE IF NOT EXISTS public.production_updates (
    id uuid DEFAULT uuid_generate_v4() PRIMARY KEY,
    project_id uuid REFERENCES public.projects(id) ON DELETE CASCADE,
    previous_etd timestamp with time zone,
    new_etd timestamp with time zone NOT NULL,
    is_on_time boolean DEFAULT true,
    delay_reason text,
    notes text,
    updated_by text,
    is_supplier_update boolean DEFAULT false,
    created_at timestamp with time zone DEFAULT now() NOT NULL
);

-- 2. Enable Row Level Security
ALTER TABLE public.production_updates ENABLE ROW LEVEL SECURITY;

-- 3. Create Access Policies
-- Allow logged-in users (PMs) full access
DROP POLICY IF EXISTS "Enable all for authenticated" ON public.production_updates;
CREATE POLICY "Enable all for authenticated" ON public.production_updates FOR ALL TO authenticated USING (true);

-- Allow public (Suppliers) to insert and read updates (Logic handled by app/RPC)
DROP POLICY IF EXISTS "Allow public insert" ON public.production_updates;
CREATE POLICY "Allow public insert" ON public.production_updates FOR INSERT TO public WITH CHECK (true);

DROP POLICY IF EXISTS "Allow public select" ON public.production_updates;
CREATE POLICY "Allow public select" ON public.production_updates FOR SELECT TO public USING (true);

-- 4. Create/Replace the Logic Function (RPC)
-- This function saves the update history AND updates the main Project ETD in one transaction.
CREATE OR REPLACE FUNCTION public.submit_production_update(
    p_project_id uuid,
    p_previous_etd timestamptz,
    p_new_etd timestamptz,
    p_is_on_time boolean,
    p_delay_reason text,
    p_notes text,
    p_updated_by text,
    p_is_supplier boolean
)
RETURNS jsonb
LANGUAGE plpgsql
SECURITY DEFINER -- Runs with admin privileges to bypass RLS on the Project Update if needed
SET search_path = public
AS $$
DECLARE
    v_record jsonb;
    v_current_milestones jsonb;
BEGIN
    -- 1. Insert the update history record
    INSERT INTO public.production_updates (
        project_id, previous_etd, new_etd, is_on_time, delay_reason, notes, updated_by, is_supplier_update
    ) VALUES (
        p_project_id, p_previous_etd, p_new_etd, p_is_on_time, p_delay_reason, p_notes, p_updated_by, p_is_supplier
    ) RETURNING to_jsonb(production_updates.*) INTO v_record;

    -- 2. Update the Project's main ETD field in the milestones JSON column
    -- Get current milestones
    SELECT milestones INTO v_current_milestones FROM public.projects WHERE id = p_project_id;
    
    IF FOUND THEN
        -- Update the 'etd' key in the jsonb object
        UPDATE public.projects 
        SET milestones = jsonb_set(
            COALESCE(v_current_milestones, '{}'::jsonb), 
            '{etd}', 
            to_jsonb(p_new_etd)
        )
        WHERE id = p_project_id;
    END IF;

    RETURN v_record;
END;
$$;

-- 5. Grant Execute Permissions
GRANT EXECUTE ON FUNCTION public.submit_production_update TO public;
GRANT EXECUTE ON FUNCTION public.submit_production_update TO authenticated;
GRANT ALL ON TABLE public.production_updates TO authenticated;
GRANT ALL ON TABLE public.production_updates TO public;

-- 6. Refresh Schema Cache
NOTIFY pgrst, 'reload schema';
