-- Migration 36: Implement PM-scoped Row Level Security (RLS) policies
-- Replaces overly permissive "Enable all for authenticated" policies with proper role-based access control
-- CRITICAL: Ensures PMs can only see projects and suppliers assigned to them

-- =====================================================
-- PROJECTS TABLE: STRICT PM ISOLATION
-- =====================================================

-- Drop existing overly permissive policy
DROP POLICY IF EXISTS "Enable all for projects" ON public.projects;

-- Admins have full access to all projects
CREATE POLICY "pm_admin_all_access_projects" ON public.projects
FOR ALL TO authenticated
USING (
  EXISTS (
    SELECT 1 FROM public.profiles
    WHERE profiles.id = auth.uid()
    AND profiles.role = 'ADMIN'
  )
);

-- PMs can only see projects where they are the assigned PM (strict isolation)
CREATE POLICY "pm_see_assigned_projects" ON public.projects
FOR SELECT TO authenticated
USING (
  EXISTS (
    SELECT 1 FROM public.profiles
    WHERE profiles.id = auth.uid()
    AND profiles.role = 'PM'
    AND projects.pm_id = auth.uid()
  )
);

-- PMs can update projects they are assigned to
CREATE POLICY "pm_update_assigned_projects" ON public.projects
FOR UPDATE TO authenticated
USING (
  EXISTS (
    SELECT 1 FROM public.profiles
    WHERE profiles.id = auth.uid()
    AND profiles.role = 'PM'
    AND projects.pm_id = auth.uid()
  )
)
WITH CHECK (
  EXISTS (
    SELECT 1 FROM public.profiles
    WHERE profiles.id = auth.uid()
    AND profiles.role = 'PM'
    AND projects.pm_id = auth.uid()
  )
);

-- PMs can create projects (they become the assigned PM and creator)
CREATE POLICY "pm_create_projects" ON public.projects
FOR INSERT TO authenticated
WITH CHECK (
  EXISTS (
    SELECT 1 FROM public.profiles
    WHERE profiles.id = auth.uid()
    AND profiles.role = 'PM'
  )
  AND created_by = auth.uid()
);

-- Only Admins can delete projects
CREATE POLICY "admin_delete_projects" ON public.projects
FOR DELETE TO authenticated
USING (
  EXISTS (
    SELECT 1 FROM public.profiles
    WHERE profiles.id = auth.uid()
    AND profiles.role = 'ADMIN'
  )
);

-- =====================================================
-- SUPPLIERS TABLE: PM-SCOPED ACCESS
-- =====================================================

-- Drop existing overly permissive policy
DROP POLICY IF EXISTS "Enable all for suppliers" ON public.suppliers;

-- Admins have full access to all suppliers
CREATE POLICY "admin_all_access_suppliers" ON public.suppliers
FOR ALL TO authenticated
USING (
  EXISTS (
    SELECT 1 FROM public.profiles
    WHERE profiles.id = auth.uid()
    AND profiles.role = 'ADMIN'
  )
);

-- PMs can see suppliers assigned to them via junction table OR suppliers used in their projects
CREATE POLICY "pm_see_assigned_suppliers" ON public.suppliers
FOR SELECT TO authenticated
USING (
  EXISTS (
    SELECT 1 FROM public.profiles prof
    WHERE prof.id = auth.uid()
    AND prof.role = 'PM'
    AND (
      -- Supplier is directly assigned to this PM
      EXISTS (
        SELECT 1 FROM public.supplier_pm_assignments spa
        WHERE spa.supplier_id = suppliers.id
        AND spa.pm_id = auth.uid()
      )
      OR
      -- Supplier is used in a project assigned to this PM
      EXISTS (
        SELECT 1 FROM public.projects p
        WHERE p.supplier_id = suppliers.id
        AND p.pm_id = auth.uid()
      )
    )
  )
);

-- PMs can update suppliers assigned to them
CREATE POLICY "pm_update_assigned_suppliers" ON public.suppliers
FOR UPDATE TO authenticated
USING (
  EXISTS (
    SELECT 1 FROM public.profiles prof
    WHERE prof.id = auth.uid()
    AND prof.role = 'PM'
    AND EXISTS (
      SELECT 1 FROM public.supplier_pm_assignments spa
      WHERE spa.supplier_id = suppliers.id
      AND spa.pm_id = auth.uid()
    )
  )
)
WITH CHECK (
  EXISTS (
    SELECT 1 FROM public.profiles prof
    WHERE prof.id = auth.uid()
    AND prof.role = 'PM'
    AND EXISTS (
      SELECT 1 FROM public.supplier_pm_assignments spa
      WHERE spa.supplier_id = suppliers.id
      AND spa.pm_id = auth.uid()
    )
  )
);

-- =====================================================
-- RFQ TABLES: PMs see only RFQs they created
-- =====================================================

DO $$
BEGIN
  IF EXISTS (SELECT 1 FROM information_schema.tables WHERE table_name = 'rfqs') THEN
    DROP POLICY IF EXISTS "Enable all for rfqs" ON public.rfqs;

    CREATE POLICY "admin_all_access_rfqs" ON public.rfqs
    FOR ALL TO authenticated
    USING (
      EXISTS (
        SELECT 1 FROM public.profiles
        WHERE profiles.id = auth.uid()
        AND profiles.role = 'ADMIN'
      )
    );

    CREATE POLICY "pm_see_own_rfqs" ON public.rfqs
    FOR SELECT TO authenticated
    USING (
      rfqs.created_by = auth.uid()
      OR EXISTS (
        SELECT 1 FROM public.profiles
        WHERE profiles.id = auth.uid()
        AND profiles.role = 'ADMIN'
      )
    );

    CREATE POLICY "pm_manage_own_rfqs" ON public.rfqs
    FOR UPDATE TO authenticated
    USING (rfqs.created_by = auth.uid());
  END IF;
END $$;
