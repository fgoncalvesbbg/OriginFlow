
-- 1. CATEGORY ATTRIBUTES TABLE
-- Drop table if it exists to ensure it's recreated with the correct schema
DROP TABLE IF EXISTS public.category_attributes;

create table public.category_attributes (
  id uuid default uuid_generate_v4() primary key,
  -- FIX: Changed category_id to uuid to match categories_l3(id) which is uuid in the live DB
  category_id uuid references public.categories_l3(id) on delete cascade,
  name text not null,
  data_type text default 'text' -- 'text' or 'number'
);

-- Enable RLS for attributes
ALTER TABLE public.category_attributes ENABLE ROW LEVEL SECURITY;
-- Admin/PM access
CREATE POLICY "Enable all for attributes" ON public.category_attributes FOR ALL USING (auth.role() = 'authenticated');
-- Public read access
CREATE POLICY "Allow public read attributes" ON public.category_attributes FOR SELECT TO public USING (true);


-- 2. UPDATE RFQS TABLE
-- Safely drop the column if it exists with the wrong type or constraint from previous failed attempts
DO $$
BEGIN
    IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'rfqs' AND column_name = 'category_id') THEN
        ALTER TABLE public.rfqs DROP COLUMN category_id;
    END IF;
END $$;

-- FIX: Changed category_id to uuid to match categories_l3(id)
ALTER TABLE public.rfqs ADD COLUMN category_id uuid references public.categories_l3(id);
ALTER TABLE public.rfqs ADD COLUMN IF NOT EXISTS attributes jsonb default '[]'::jsonb;

-- Force schema cache reload to recognize the new Foreign Keys immediately
NOTIFY pgrst, 'reload schema';
