
-- 1. Ensure Authenticated Users can Insert into Steps and Documents
-- This fixes the issue where new projects have empty checklists because the seed insert fails silently due to RLS.

ALTER TABLE public.project_steps ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "Enable all for steps" ON public.project_steps;
CREATE POLICY "Enable all for steps" ON public.project_steps FOR ALL USING (auth.role() = 'authenticated');

ALTER TABLE public.project_documents ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "Enable all for docs" ON public.project_documents;
CREATE POLICY "Enable all for docs" ON public.project_documents FOR ALL USING (auth.role() = 'authenticated');

-- 2. Ensure Production Updates table is writable by authenticated users
ALTER TABLE public.production_updates ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "Enable all for authenticated" ON public.production_updates;
CREATE POLICY "Enable all for authenticated" ON public.production_updates FOR ALL TO authenticated USING (true);

-- 3. Refresh schema to apply changes
NOTIFY pgrst, 'reload schema';
