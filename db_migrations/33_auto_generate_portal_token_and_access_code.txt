
-- 1. ADD DEFAULT FOR PORTAL_TOKEN (UUID)
-- Rename 'token' column to 'portal_token' if it exists, otherwise just add default
DO $$
BEGIN
  -- Check if old 'token' column exists
  IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'suppliers' AND column_name = 'token') THEN
    -- Rename token to portal_token
    ALTER TABLE suppliers RENAME COLUMN token TO portal_token;
  END IF;

  -- Set default for portal_token
  ALTER TABLE suppliers ALTER COLUMN portal_token SET DEFAULT gen_random_uuid()::text;

  -- Backfill existing suppliers without portal_token
  UPDATE suppliers SET portal_token = gen_random_uuid()::text WHERE portal_token IS NULL;

  -- Make portal_token NOT NULL
  ALTER TABLE suppliers ALTER COLUMN portal_token SET NOT NULL;
EXCEPTION WHEN OTHERS THEN
  -- Column may already be properly configured
  NULL;
END $$;

-- 2. CREATE FUNCTION TO GENERATE 6-DIGIT ACCESS CODE
CREATE OR REPLACE FUNCTION generate_access_code()
RETURNS TEXT AS $$
BEGIN
  -- Generate random 6-digit number padded with zeros
  RETURN LPAD(FLOOR(RANDOM() * 1000000)::TEXT, 6, '0');
END;
$$ LANGUAGE plpgsql;

-- 3. CREATE TRIGGER FUNCTION TO AUTO-GENERATE ACCESS CODE
CREATE OR REPLACE FUNCTION ensure_access_code()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.access_code IS NULL OR NEW.access_code = '' THEN
    NEW.access_code := generate_access_code();
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 4. CREATE TRIGGER (DROP IF EXISTS FIRST TO AVOID CONFLICTS)
DROP TRIGGER IF EXISTS supplier_ensure_access_code ON suppliers;
CREATE TRIGGER supplier_ensure_access_code
  BEFORE INSERT ON suppliers
  FOR EACH ROW
  EXECUTE FUNCTION ensure_access_code();

-- 5. BACKFILL EXISTING SUPPLIERS WITHOUT ACCESS CODE
UPDATE suppliers
SET access_code = generate_access_code()
WHERE access_code IS NULL OR access_code = '';

-- 6. MAKE ACCESS CODE NOT NULL
ALTER TABLE suppliers ALTER COLUMN access_code SET NOT NULL;

-- 7. REFRESH SCHEMA CACHE
NOTIFY pgrst, 'reload schema';
