

-- Enable UUID extension
create extension if not exists "uuid-ossp";

ALTER TABLE public.compliance_requirements ADD COLUMN IF NOT EXISTS section text;
-- --------------------------------------------------

-- 1. PROFILES (Users)
create table if not exists public.profiles (
  id uuid references auth.users on delete cascade not null primary key,
  email text not null,
  name text,
  role text default 'PM', -- 'PM' or 'ADMIN'
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 2. SUPPLIERS
create table if not exists public.suppliers (
  id uuid default uuid_generate_v4() primary key,
  name text not null,
  code text,
  email text,
  token text, 
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 3. PROJECTS
create table if not exists public.projects (
  id uuid default uuid_generate_v4() primary key,
  project_id_code text not null, -- e.g. PRJ-2025-001
  name text not null,
  supplier_id uuid references public.suppliers(id),
  pm_id uuid references public.profiles(id),
  current_step integer default 1,
  status text default 'idea',
  supplier_link_token text,
  milestones jsonb default '{}'::jsonb,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 4. PROJECT STEPS
create table if not exists public.project_steps (
  id uuid default uuid_generate_v4() primary key,
  project_id uuid references public.projects(id) on delete cascade,
  step_number integer not null,
  name text not null,
  status text default 'not_started'
);

-- 5. PROJECT DOCUMENTS
create table if not exists public.project_documents (
  id uuid default uuid_generate_v4() primary key,
  project_id uuid references public.projects(id) on delete cascade,
  step_number integer not null,
  title text not null,
  description text,
  responsible_party text default 'internal', -- 'internal' or 'supplier'
  is_visible_to_supplier boolean default true,
  is_required boolean default true,
  status text default 'not_started',
  file_url text,
  deadline timestamp with time zone,
  uploaded_at timestamp with time zone,
  uploaded_by_supplier boolean default false,
  supplier_comment text,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 6. DOCUMENT VERSIONS
create table if not exists public.document_versions (
  id uuid default uuid_generate_v4() primary key,
  document_id uuid references public.project_documents(id) on delete cascade,
  file_url text not null,
  version_number integer not null,
  uploaded_by_supplier boolean default false,
  uploaded_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 7. NOTIFICATIONS
create table if not exists public.notifications (
  id uuid default uuid_generate_v4() primary key,
  user_id uuid references public.profiles(id),
  supplier_id uuid references public.suppliers(id),
  message text not null,
  link text,
  is_read boolean default false,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 8. DOCUMENT COMMENTS (Contextual Chat)
create table if not exists public.document_comments (
  id uuid default uuid_generate_v4() primary key,
  document_id uuid references public.project_documents(id) on delete cascade,
  author_name text not null,
  author_role text not null, -- 'PM', 'ADMIN', 'SUPPLIER'
  content text not null,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- --- COMPLIANCE MODULE ---

-- 9. CATEGORIES (Level 3)
create table if not exists public.categories_l3 (
  id text primary key, -- e.g., 'cat-123'
  name text not null,
  description text,
  active boolean default true,
  is_finalized boolean default false,
  finalized_at timestamp with time zone
);

-- 10. PRODUCT FEATURES
create table if not exists public.product_features (
  id text primary key, -- e.g. 'feat-abc'
  category_id text references public.categories_l3(id) on delete cascade,
  name text not null,
  description text,
  active boolean default true
);

-- 11. COMPLIANCE REQUIREMENTS
create table if not exists public.compliance_requirements (
  id uuid primary key, -- Changed from text to uuid
  category_id text references public.categories_l3(id) on delete cascade,
  section text, -- Section grouping (e.g. "Electrical Test Reports")
  title text not null,
  description text,
  is_mandatory boolean default true,
  reference_code text,
  applies_by_default boolean default true,
  condition_feature_ids text[] default '{}'::text[] -- Array of feature IDs
);

-- 12. COMPLIANCE REQUESTS (TCF)
create table if not exists public.compliance_requests (
  id uuid default uuid_generate_v4() primary key,
  request_id text not null, -- Business ID e.g. TCF-2025-001
  project_id uuid references public.projects(id) on delete set null,
  project_name text not null,
  supplier_id uuid references public.suppliers(id),
  category_id text references public.categories_l3(id),
  features jsonb default '[]'::jsonb, -- Snapshot of features {featureId, value}
  status text default 'pending_supplier',
  responses jsonb default '[]'::jsonb, -- User answers
  token text, -- Access token for supplier
  deadline timestamp with time zone,
  submitted_at timestamp with time zone,
  completed_at timestamp with time zone,
  updated_by text,
  change_log jsonb default '[]'::jsonb,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- --- INSTRUCTION MANUAL (IM) MODULE ---

-- 13. IM TEMPLATES
create table if not exists public.im_templates (
  id uuid default uuid_generate_v4() primary key,
  category_id text references public.categories_l3(id) on delete cascade,
  name text not null,
  is_in_design_implemented boolean default false,
  languages text[] default '{en}'::text[],
  is_finalized boolean default false,
  finalized_at timestamp with time zone,
  metadata jsonb default '{}'::jsonb, -- Covers, logos, colors
  last_updated_by text,
  updated_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 14. IM SECTIONS
create table if not exists public.im_sections (
  id uuid default uuid_generate_v4() primary key,
  template_id uuid references public.im_templates(id) on delete cascade,
  parent_id uuid references public.im_sections(id) on delete cascade, -- For subsections
  title text not null,
  "order" integer default 0,
  is_placeholder boolean default false,
  content jsonb default '{}'::jsonb, -- Keyed by language code: { "en": "html..." }
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 15. PROJECT IM INSTANCES
create table if not exists public.project_ims (
  id uuid default uuid_generate_v4() primary key,
  project_id uuid references public.projects(id) on delete cascade,
  template_id uuid references public.im_templates(id),
  status text default 'draft', -- 'draft', 'generated'
  placeholder_data jsonb default '{}'::jsonb, -- User inputs for placeholders
  updated_at timestamp with time zone default timezone('utc'::text, now()) not null
);
